@dataclass

    pattern_name: str  # 패턴 이름 (예: "Ascending Triangle")
    confidence_score: float  # 신뢰도 0-100점
    entry_price: float  # 권장 진입가격
    target_price: float  # 목표가격
    stop_loss_price: float  # 손절가격
    expected_return: float  # 기대수익률 (%)
    success_probability: float  # 역사적 성공률 (%)
    volume_confirmation: bool  # 거래량 확인 여부
    reasoning: str  # 분석 근거 설명
    detected_at: datetime  # 패턴 감지 시점

class BulkowskiPatternAnalyzer:
    """토마스 불코우스키 차트 패턴 분석기"""

    def __init__(self):
        # 불코우스키의 패턴별 통계적 성공률 데이터
        self.pattern_success_rates = {
            "ascending_triangle": 72.0,  # 상승삼각형 성공률 72%
            "descending_triangle": 68.0,  # 하락삼각형 성공률 68%
            "symmetrical_triangle": 65.0,  # 대칭삼각형 성공률 65%
            "double_bottom": 78.0,  # 더블바텀 성공률 78%
            "double_top": 73.0,  # 더블탑 성공률 73%
            "head_shoulders": 81.0,  # 헤드앤숄더 성공률 81%
            "cup_handle": 76.0,  # 컵앤핸들 성공률 76%
            "flag": 69.0,  # 플래그 패턴 성공률 69%
            "pennant": 71.0,  # 페넌트 패턴 성공률 71%
        }

    def analyze_chart_patterns(self, df: pd.DataFrame) -> List[ChartPatternResult]:
        """
        토마스 불코우스키 방식의 차트 패턴 분석

        Args:
            df: OHLCV 데이터프레임

        Returns:
            감지된 차트 패턴 리스트 (Gemini AI가 쉽게 이해할 수 있는 형태)
        """
        patterns = []

        try:
            # 1. 상승삼각형 패턴 분석
            ascending_triangle = self._detect_ascending_triangle(df)
            if ascending_triangle:
                patterns.append(ascending_triangle)

            # 2. 하락삼각형 패턴 분석
            descending_triangle = self._detect_descending_triangle(df)
            if descending_triangle:
                patterns.append(descending_triangle)

            # 3. 더블바텀 패턴 분석
            double_bottom = self._detect_double_bottom(df)
            if double_bottom:
                patterns.append(double_bottom)

            # 4. 더블탑 패턴 분석
            double_top = self._detect_double_top(df)
            if double_top:
                patterns.append(double_top)

            # 5. 헤드앤숄더 패턴 분석
            head_shoulders = self._detect_head_shoulders(df)
            if head_shoulders:
                patterns.append(head_shoulders)

            logger.info(f"불코우스키 패턴 분석 완료: {len(patterns)}개 패턴 감지")

        except Exception as e:
            logger.error(f"불코우스키 패턴 분석 실패: {e}")

        return patterns

    def _detect_ascending_triangle(
        self, df: pd.DataFrame
    ) -> Optional[ChartPatternResult]:
        """상승삼각형 패턴 감지 (불코우스키 기법)"""
        try:
            if len(df) < 20:
                return None

            recent_data = df.tail(20)
            highs = recent_data["high"].values
            lows = recent_data["low"].values
            closes = recent_data["close"].values
            volumes = recent_data["volume"].values

            # 불코우스키 조건: 수평 저항선 + 상승 지지선
            resistance_level = np.max(highs[-10:])  # 최근 10일 최고가를 저항선으로

            # 상승 지지선 검증 (최소 3개 저점이 상승 패턴)
            support_points = []
            for i in range(len(lows) - 2):
                if lows[i - 1] > lows[i] < lows[i + 1]:  # 저점 감지
                    support_points.append((i, lows[i]))

            if len(support_points) < 3:
                return None

            # 지지선이 상승하는지 확인
            support_prices = [point[1] for point in support_points[-3:]]
            if not (support_prices[0] < support_prices[1] < support_prices[2]):
                return None

            # 거래량 확인 (불코우스키 중요 조건)
            avg_volume = np.mean(volumes[:-3])
            recent_volume = np.mean(volumes[-3:])
            volume_confirmation = recent_volume > avg_volume * 1.2

            # 목표가 계산 (불코우스키 공식: 패턴 높이를 돌파점에 더함)
            pattern_height = resistance_level - min(support_prices)
            entry_price = resistance_level * 1.02  # 저항선 돌파 확인 후 진입
            target_price = resistance_level + pattern_height
            stop_loss = min(support_prices) * 0.98

            # 수익률 계산
            expected_return = ((target_price - entry_price) / entry_price) * 100

            # 신뢰도 계산
            base_confidence = self.pattern_success_rates["ascending_triangle"]
            volume_bonus = 10 if volume_confirmation else -10
            pattern_clarity = self._calculate_pattern_clarity(
                support_points, resistance_level
            )
            confidence_score = min(
                95, max(30, base_confidence + volume_bonus + pattern_clarity)
            )

            return ChartPatternResult(
                pattern_name="상승삼각형 (Ascending Triangle)",
                confidence_score=confidence_score,
                entry_price=entry_price,
                target_price=target_price,
                stop_loss_price=stop_loss,
                expected_return=expected_return,
                success_probability=self.pattern_success_rates["ascending_triangle"],
                volume_confirmation=volume_confirmation,
                reasoning=f"수평 저항선 {resistance_level:.0f}원과 상승 지지선 형성. "
                f"패턴 높이 {pattern_height:.0f}원을 바탕으로 목표가 {target_price:.0f}원 설정. "
                f"거래량 {'확인됨' if volume_confirmation else '미확인'}.",
                detected_at=datetime.now(),
            )

        except Exception as e:
            logger.error(f"상승삼각형 감지 실패: {e}")
            return None

    def _detect_descending_triangle(
        self, df: pd.DataFrame
    ) -> Optional[ChartPatternResult]:
        """하락삼각형 패턴 감지"""
        try:
            if len(df) < 20:
                return None

            recent_data = df.tail(20)
            highs = recent_data["high"].values
            lows = recent_data["low"].values
            closes = recent_data["close"].values
            volumes = recent_data["volume"].values

            # 수평 지지선 + 하락 저항선
            support_level = np.min(lows[-10:])

            # 하락 저항선 검증
            peak_points = []
            for i in range(1, len(highs) - 1):
                if highs[i - 1] < highs[i] > highs[i + 1]:
                    peak_points.append((i, highs[i]))

            if len(peak_points) < 3:
                return None

            peak_prices = [point[1] for point in peak_points[-3:]]
            if not (peak_prices[0] > peak_prices[1] > peak_prices[2]):
                return None

            # 거래량 및 목표가 계산
            avg_volume = np.mean(volumes[:-3])
            recent_volume = np.mean(volumes[-3:])
            volume_confirmation = recent_volume > avg_volume * 1.2

            pattern_height = max(peak_prices) - support_level
            entry_price = support_level * 0.98  # 지지선 이탈 확인 후 진입
            target_price = support_level - pattern_height
            stop_loss = max(peak_prices) * 1.02

            expected_return = (
                (entry_price - target_price) / entry_price
            ) * 100  # 공매도 수익률

            base_confidence = self.pattern_success_rates["descending_triangle"]
            volume_bonus = 10 if volume_confirmation else -10
            confidence_score = min(95, max(30, base_confidence + volume_bonus))

            return ChartPatternResult(
                pattern_name="하락삼각형 (Descending Triangle)",
                confidence_score=confidence_score,
                entry_price=entry_price,
                target_price=target_price,
                stop_loss_price=stop_loss,
                expected_return=expected_return,
                success_probability=self.pattern_success_rates["descending_triangle"],
                volume_confirmation=volume_confirmation,
                reasoning=f"수평 지지선 {support_level:.0f}원과 하락 저항선 형성. "
                f"하락 목표가 {target_price:.0f}원. "
                f"거래량 {'확인됨' if volume_confirmation else '미확인'}.",
                detected_at=datetime.now(),
            )

        except Exception as e:
            logger.error(f"하락삼각형 감지 실패: {e}")
            return None

    def _detect_double_bottom(self, df: pd.DataFrame) -> Optional[ChartPatternResult]:
        """더블바텀 패턴 감지 (불코우스키 기법)"""
        try:
            if len(df) < 30:
                return None

            recent_data = df.tail(30)
            lows = recent_data["low"].values
            closes = recent_data["close"].values
            volumes = recent_data["volume"].values

            # 두 개의 저점 찾기
            bottom_points = []
            for i in range(2, len(lows) - 2):
                if lows[i - 2] > lows[i - 1] > lows[i] < lows[i + 1] < lows[i + 2]:
                    bottom_points.append((i, lows[i]))

            if len(bottom_points) < 2:
                return None

            # 가장 최근 두 저점 선택
            first_bottom = bottom_points[-2]
            second_bottom = bottom_points[-1]

            # 두 저점이 비슷한 수준인지 확인 (5% 이내 차이)
            price_diff = abs(first_bottom[1] - second_bottom[1]) / min(
                first_bottom[1], second_bottom[1]
            )
            if price_diff > 0.05:
                return None

            # 중간 고점 찾기
            middle_section = lows[first_bottom[0] : second_bottom[0]]
            if len(middle_section) < 3:
                return None

            middle_high = np.max(middle_section)
            bottom_level = (first_bottom[1] + second_bottom[1]) / 2

            # 패턴 높이와 목표가 계산
            pattern_height = middle_high - bottom_level
            entry_price = middle_high * 1.01  # 목선(neckline) 돌파 확인
            target_price = middle_high + pattern_height
            stop_loss = bottom_level * 0.97

            # 거래량 확인
            first_volume = volumes[first_bottom[0]]
            second_volume = volumes[second_bottom[0]]
            volume_confirmation = (
                second_volume < first_volume * 0.8
            )  # 두 번째 저점에서 거래량 감소

            expected_return = ((target_price - entry_price) / entry_price) * 100

            base_confidence = self.pattern_success_rates["double_bottom"]
            volume_bonus = 10 if volume_confirmation else -5
            confidence_score = min(95, max(30, base_confidence + volume_bonus))

            return ChartPatternResult(
                pattern_name="더블바텀 (Double Bottom)",
                confidence_score=confidence_score,
                entry_price=entry_price,
                target_price=target_price,
                stop_loss_price=stop_loss,
                expected_return=expected_return,
                success_probability=self.pattern_success_rates["double_bottom"],
                volume_confirmation=volume_confirmation,
                reasoning=f"두 저점 {bottom_level:.0f}원 형성, 목선 {middle_high:.0f}원 돌파 시 "
                f"목표가 {target_price:.0f}원. 두 번째 저점 거래량 {'감소 확인' if volume_confirmation else '미확인'}.",
                detected_at=datetime.now(),
            )

        except Exception as e:
            logger.error(f"더블바텀 감지 실패: {e}")
            return None

    def _detect_double_top(self, df: pd.DataFrame) -> Optional[ChartPatternResult]:
        """더블탑 패턴 감지"""
        try:
            if len(df) < 30:
                return None

            recent_data = df.tail(30)
            highs = recent_data["high"].values
            closes = recent_data["close"].values
            volumes = recent_data["volume"].values

            # 두 개의 고점 찾기
            peak_points = []
            for i in range(2, len(highs) - 2):
                if highs[i - 2] < highs[i - 1] < highs[i] > highs[i + 1] > highs[i + 2]:
                    peak_points.append((i, highs[i]))

            if len(peak_points) < 2:
                return None

            first_peak = peak_points[-2]
            second_peak = peak_points[-1]

            # 두 고점이 비슷한 수준인지 확인
            price_diff = abs(first_peak[1] - second_peak[1]) / max(
                first_peak[1], second_peak[1]
            )
            if price_diff > 0.05:
                return None

            # 중간 저점 찾기
            middle_section = highs[first_peak[0] : second_peak[0]]
            middle_low = np.min(middle_section)
            peak_level = (first_peak[1] + second_peak[1]) / 2

            pattern_height = peak_level - middle_low
            entry_price = middle_low * 0.99  # 목선 이탈 확인
            target_price = middle_low - pattern_height
            stop_loss = peak_level * 1.03

            # 거래량 확인 (두 번째 고점에서 거래량 감소)
            first_volume = volumes[first_peak[0]]
            second_volume = volumes[second_peak[0]]
            volume_confirmation = second_volume < first_volume * 0.8

            expected_return = ((entry_price - target_price) / entry_price) * 100

            base_confidence = self.pattern_success_rates["double_top"]
            volume_bonus = 10 if volume_confirmation else -5
            confidence_score = min(95, max(30, base_confidence + volume_bonus))

            return ChartPatternResult(
                pattern_name="더블탑 (Double Top)",
                confidence_score=confidence_score,
                entry_price=entry_price,
                target_price=target_price,
                stop_loss_price=stop_loss,
                expected_return=expected_return,
                success_probability=self.pattern_success_rates["double_top"],
                volume_confirmation=volume_confirmation,
                reasoning=f"두 고점 {peak_level:.0f}원 형성, 목선 {middle_low:.0f}원 이탈 시 "
                f"하락 목표가 {target_price:.0f}원. 두 번째 고점 거래량 {'감소 확인' if volume_confirmation else '미확인'}.",
                detected_at=datetime.now(),
            )

        except Exception as e:
            logger.error(f"더블탑 감지 실패: {e}")
            return None

    def _detect_head_shoulders(self, df: pd.DataFrame) -> Optional[ChartPatternResult]:
        """헤드앤숄더 패턴 감지"""
        try:
            if len(df) < 40:
                return None

            recent_data = df.tail(40)
            highs = recent_data["high"].values
            volumes = recent_data["volume"].values

            # 세 개의 고점 찾기 (왼쪽 어깨, 머리, 오른쪽 어깨)
            peak_points = []
            for i in range(3, len(highs) - 3):
                if (
                    highs[i - 3]
                    < highs[i - 2]
                    < highs[i - 1]
                    < highs[i]
                    > highs[i + 1]
                    > highs[i + 2]
                    > highs[i + 3]
                ):
                    peak_points.append((i, highs[i]))

            if len(peak_points) < 3:
                return None

            # 최근 3개 고점을 어깨-머리-어깨로 가정
            left_shoulder = peak_points[-3]
            head = peak_points[-2]
            right_shoulder = peak_points[-1]

            # 헤드앤숄더 조건 확인
            # 1. 머리가 가장 높아야 함
            if not (head[1] > left_shoulder[1] and head[1] > right_shoulder[1]):
                return None

            # 2. 두 어깨가 비슷한 높이여야 함 (10% 이내)
            shoulder_diff = abs(left_shoulder[1] - right_shoulder[1]) / max(
                left_shoulder[1], right_shoulder[1]
            )
            if shoulder_diff > 0.10:
                return None

            # 목선 계산 (두 어깨 사이의 저점들 연결)
            neckline_section = highs[left_shoulder[0] : right_shoulder[0]]
            neckline_level = np.min(neckline_section)

            # 패턴 높이와 목표가
            pattern_height = head[1] - neckline_level
            entry_price = neckline_level * 0.99
            target_price = neckline_level - pattern_height
            stop_loss = min(left_shoulder[1], right_shoulder[1]) * 1.02

            # 거래량 패턴 확인 (이상적으로는 왼쪽 어깨 > 머리 > 오른쪽 어깨)
            left_vol = volumes[left_shoulder[0]]
            head_vol = volumes[head[0]]
            right_vol = volumes[right_shoulder[0]]
            volume_confirmation = left_vol > head_vol > right_vol

            expected_return = ((entry_price - target_price) / entry_price) * 100

            base_confidence = self.pattern_success_rates["head_shoulders"]
            volume_bonus = 15 if volume_confirmation else -5
            confidence_score = min(95, max(30, base_confidence + volume_bonus))

            return ChartPatternResult(
                pattern_name="헤드앤숄더 (Head and Shoulders)",
                confidence_score=confidence_score,
                entry_price=entry_price,
                target_price=target_price,
                stop_loss_price=stop_loss,
                expected_return=expected_return,
                success_probability=self.pattern_success_rates["head_shoulders"],
                volume_confirmation=volume_confirmation,
                reasoning=f"머리 {head[1]:.0f}원, 어깨 {(left_shoulder[1]+right_shoulder[1])/2:.0f}원, "
                f"목선 {neckline_level:.0f}원. 목선 이탈 시 목표가 {target_price:.0f}원. "
                f"거래량 패턴 {'확인' if volume_confirmation else '미확인'}.",
                detected_at=datetime.now(),
            )

        except Exception as e:
            logger.error(f"헤드앤숄더 감지 실패: {e}")
            return None

    def _calculate_pattern_clarity(self, points: List, reference_level: float) -> float:
        """패턴의 명확성 점수 계산 (-10 ~ +10)"""
        try:
            if len(points) < 2:
                return -10

            # 점들이 기준선에서 얼마나 일관되게 떨어져 있는지 측정
            distances = [
                abs(point[1] - reference_level) / reference_level for point in points
            ]
            consistency = 1.0 - np.std(distances)

            return min(10, max(-10, consistency * 20 - 10))

        except:
            return 0


class TechnicalAnalyzer:
    """
    차트 전문가 기술적 분석 시스템 메인 클래스
    토마스 불코우스키 + 알렉산더 엘더 기법 통합
    Gemini AI 최적화 설계
    """

    def __init__(self):
        self.bulkowski_analyzer = BulkowskiPatternAnalyzer()
        self.elder_analyzer = ElderTripleScreenAnalyzer()
        self.logger = logging.getLogger(__name__)

    async def analyze_all_markets(
        self, market_data: Dict[str, List[StockData]]
    ) -> Dict[str, Any]:
        """
        전체 시장 기술적 분석 (차트 전문가 기법 적용)

        Args:
            market_data: 시장별 종목 데이터

        Returns:
            Gemini AI가 이해하기 쉬운 종합 분석 결과
        """
        analysis_results = {}

        try:
            for market_name, stocks in market_data.items():
                self.logger.info(f"{market_name} 시장 차트 전문가 분석 시작...")

                market_analysis = {
                    "market_name": market_name,
                    "total_stocks": len(stocks),
                    "analyzed_stocks": [],
                    "chart_patterns_summary": {},
                    "triple_screen_summary": {},
                    "top_opportunities": [],
                }

                for stock_data_list in stocks:
                    if isinstance(stock_data_list, list) and len(stock_data_list) > 0:
                        stock_analysis = await self.analyze_single_stock(
                            stock_data_list
                        )
                        if stock_analysis:
                            market_analysis["analyzed_stocks"].append(stock_analysis)

                # 시장별 요약 생성
                market_analysis["chart_patterns_summary"] = (
                    self._summarize_chart_patterns(market_analysis["analyzed_stocks"])
                )
                market_analysis["triple_screen_summary"] = (
                    self._summarize_triple_screen_results(
                        market_analysis["analyzed_stocks"]
                    )
                )
                market_analysis["top_opportunities"] = self._identify_top_opportunities(
                    market_analysis["analyzed_stocks"]
                )

                analysis_results[market_name] = market_analysis

                self.logger.info(
                    f"{market_name} 분석 완료: {len(market_analysis['analyzed_stocks'])}개 종목"
                )

        except Exception as e:
            self.logger.error(f"전체 시장 분석 실패: {e}")

        return analysis_results

    async def analyze_single_stock(
        self, stock_data: List[StockData]
    ) -> Optional[Dict[str, Any]]:
        """
        개별 종목 차트 전문가 분석

        Args:
            stock_data: 개별 종목의 시계열 데이터

        Returns:
            종목별 분석 결과 (Gemini AI 친화적 형태)
        """
        try:
            if not stock_data or len(stock_data) < 20:
                return None

            # 데이터프레임 변환
            df = self._convert_to_dataframe(stock_data)
            if df.empty:
                return None

            symbol = stock_data[0].symbol
            company_name = getattr(stock_data[0], "company_name", symbol)
            current_price = df["close"].iloc[-1]

            # 1. 불코우스키 차트 패턴 분석
            chart_patterns = self.bulkowski_analyzer.analyze_chart_patterns(df)

            # 2. 엘더 삼중스크린 분석
            triple_screen = self.elder_analyzer.analyze_triple_screen(df)

            # 3. 기본 기술적 지표 계산
            basic_indicators = self._calculate_basic_indicators(df)

            # 4. 종합 투자 신호 생성
            investment_signal = self._generate_investment_signal(
                chart_patterns, triple_screen, basic_indicators
            )

            # 5. Gemini AI를 위한 분석 요약
            analysis_summary = self._create_gemini_friendly_summary(
                symbol,
                company_name,
                current_price,
                chart_patterns,
                triple_screen,
                basic_indicators,
                investment_signal,
            )

            return {
                "symbol": symbol,
                "company_name": company_name,
                "current_price": current_price,
                "chart_patterns": [pattern.__dict__ for pattern in chart_patterns],
                "triple_screen": triple_screen.__dict__,
                "basic_indicators": basic_indicators,
                "investment_signal": investment_signal,
                "gemini_summary": analysis_summary,
                "analysis_timestamp": datetime.now(),
            }

        except Exception as e:
            self.logger.error(f"개별 종목 분석 실패: {e}")
            return None

    def _convert_to_dataframe(self, stock_data: List[StockData]) -> pd.DataFrame:
        """StockData 리스트를 DataFrame으로 변환"""
        try:
            data_list = []
            for data in stock_data:
                data_list.append(
                    {
                        "date": data.date,
                        "open": data.open_price,
                        "high": data.high_price,
                        "low": data.low_price,
                        "close": data.close_price,
                        "volume": data.volume,
                    }
                )

            df = pd.DataFrame(data_list)
            df["date"] = pd.to_datetime(df["date"])
            df.set_index("date", inplace=True)
            df.sort_index(inplace=True)

            # 데이터 검증
            df = df.dropna()
            df = df[df["volume"] > 0]  # 거래량 0인 데이터 제거

            return df

        except Exception as e:
            self.logger.error(f"DataFrame 변환 실패: {e}")
            return pd.DataFrame()

    def _calculate_basic_indicators(self, df: pd.DataFrame) -> Dict[str, Any]:
        """기본 기술적 지표 계산 (Gemini AI 이해용)"""
        try:
            indicators = {}

            # 이동평균선
            indicators["ma_5"] = df["close"].rolling(5).mean().iloc[-1]
            indicators["ma_20"] = df["close"].rolling(20).mean().iloc[-1]
            indicators["ma_60"] = df["close"].rolling(60).mean().iloc[-1]

            # RSI
            delta = df["close"].diff()
            gain = (delta.where(delta > 0, 0)).rolling(14).mean()
            loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
            rs = gain / loss
            rsi = 100 - (100 / (1 + rs))
            indicators["rsi"] = rsi.iloc[-1]

            # 볼린저밴드
            bb_middle = df["close"].rolling(20).mean()
            bb_std = df["close"].rolling(20).std()
            indicators["bb_upper"] = (bb_middle + 2 * bb_std).iloc[-1]
            indicators["bb_lower"] = (bb_middle - 2 * bb_std).iloc[-1]
            indicators["bb_position"] = (
                df["close"].iloc[-1] - indicators["bb_lower"]
            ) / (indicators["bb_upper"] - indicators["bb_lower"])

            # MACD
            ema_12 = df["close"].ewm(span=12).mean()
            ema_26 = df["close"].ewm(span=26).mean()
            macd_line = ema_12 - ema_26
            signal_line = macd_line.ewm(span=9).mean()
            indicators["macd"] = macd_line.iloc[-1]
            indicators["macd_signal"] = signal_line.iloc[-1]
            indicators["macd_histogram"] = (macd_line - signal_line).iloc[-1]

            # 거래량 분석
            avg_volume_20 = df["volume"].rolling(20).mean().iloc[-1]
            current_volume = df["volume"].iloc[-1]
            indicators["volume_ratio"] = current_volume / avg_volume_20

            return indicators

        except Exception as e:
            self.logger.error(f"기본 지표 계산 실패: {e}")
            return {}

    def _generate_investment_signal(
        self,
        chart_patterns: List[ChartPatternResult],
        triple_screen: TripleScreenResult,
        basic_indicators: Dict[str, Any],
    ) -> Dict[str, Any]:
        """종합 투자 신호 생성"""
        try:
            signal_strength = 0
            signal_type = "HOLD"
            confidence = 50
            reasons = []

            # 차트 패턴 점수
            pattern_score = 0
            if chart_patterns:
                best_pattern = max(chart_patterns, key=lambda x: x.confidence_score)
                pattern_score = best_pattern.confidence_score
                if best_pattern.expected_return > 5:
                    signal_strength += 20
                    reasons.append(f"{best_pattern.pattern_name} 패턴 감지")

            # 삼중스크린 점수
            triple_score = triple_screen.overall_score
            if triple_screen.final_signal == "강력한 매수":
                signal_strength += 30
                reasons.append("엘더 삼중스크린 매수 신호")
            elif triple_screen.final_signal == "강력한 매도":
                signal_strength -= 30
                reasons.append("엘더 삼중스크린 매도 신호")

            # 기본 지표 점수
            rsi = basic_indicators.get("rsi", 50)
            if rsi < 30:
                signal_strength += 15
                reasons.append("RSI 과매도")
            elif rsi > 70:
                signal_strength -= 15
                reasons.append("RSI 과매수")

            # 최종 신호 결정
            if signal_strength >= 40:
                signal_type = "STRONG_BUY"
                confidence = min(95, 70 + signal_strength / 2)
            elif signal_strength >= 20:
                signal_type = "BUY"
                confidence = min(85, 60 + signal_strength / 2)
            elif signal_strength <= -40:
                signal_type = "STRONG_SELL"
                confidence = min(95, 70 + abs(signal_strength) / 2)
            elif signal_strength <= -20:
                signal_type = "SELL"
                confidence = min(85, 60 + abs(signal_strength) / 2)

            return {
                "signal_type": signal_type,
                "signal_strength": signal_strength,
                "confidence": confidence,
                "pattern_score": pattern_score,
                "triple_screen_score": triple_score,
                "rsi_score": rsi,
                "reasons": reasons,
            }

        except Exception as e:
            self.logger.error(f"투자 신호 생성 실패: {e}")
            return {
                "signal_type": "HOLD",
                "signal_strength": 0,
                "confidence": 50,
                "reasons": [],
            }

    def _create_gemini_friendly_summary(
        self,
        symbol: str,
        company_name: str,
        current_price: float,
        chart_patterns: List[ChartPatternResult],
        triple_screen: TripleScreenResult,
        basic_indicators: Dict[str, Any],
        investment_signal: Dict[str, Any],
    ) -> str:
        """Gemini AI가 이해하기 쉬운 분석 요약 생성"""
        try:
            summary = f"【{symbol} ({company_name}) 차트 전문가 분석】\n"
            summary += f"현재가: {current_price:,.0f}원\n\n"

            # 차트 패턴 요약
            if chart_patterns:
                summary += "🔍 불코우스키 차트 패턴:\n"
                for pattern in chart_patterns:
                    summary += f"  • {pattern.pattern_name}: 신뢰도 {pattern.confidence_score:.0f}%, "
                    summary += f"목표가 {pattern.target_price:,.0f}원 (수익률 {pattern.expected_return:.1f}%)\n"
                    summary += f"    진입가 {pattern.entry_price:,.0f}원, 손절가 {pattern.stop_loss_price:,.0f}원\n"
            else:
                summary += "🔍 불코우스키 차트 패턴: 감지된 패턴 없음\n"

            # 삼중스크린 요약
            summary += f"\n📊 엘더 삼중스크린 시스템:\n"
            summary += f"  • 1차스크린(장기추세): {triple_screen.screen1_trend} ({triple_screen.screen1_score:.0f}점)\n"
            summary += f"  • 2차스크린(오실레이터): {triple_screen.screen2_oscillator} ({triple_screen.screen2_score:.0f}점)\n"
            summary += f"  • 3차스크린(진입타이밍): {triple_screen.screen3_timing} ({triple_screen.screen3_score:.0f}점)\n"
            summary += f"  • 최종신호: {triple_screen.final_signal} (종합점수 {triple_screen.overall_score:.0f}점)\n"

            # 기본 지표 요약
            summary += f"\n📈 주요 기술적 지표:\n"
            summary += f"  • RSI: {basic_indicators.get('rsi', 0):.1f}\n"
            summary += f"  • 이동평균: 5일 {basic_indicators.get('ma_5', 0):,.0f}, 20일 {basic_indicators.get('ma_20', 0):,.0f}원\n"
            summary += (
                f"  • 거래량 비율: {basic_indicators.get('volume_ratio', 1):.1f}배\n"
            )

            # 투자 신호 요약
            summary += f"\n🎯 종합 투자 신호: {investment_signal['signal_type']} "
            summary += f"(신뢰도 {investment_signal['confidence']:.0f}%)\n"
            if investment_signal["reasons"]:
                summary += f"  • 근거: {', '.join(investment_signal['reasons'])}\n"

            return summary

        except Exception as e:
            self.logger.error(f"Gemini 요약 생성 실패: {e}")
            return f"{symbol} 분석 요약 생성 중 오류 발생"

    def _summarize_chart_patterns(self, analyzed_stocks: List[Dict]) -> Dict[str, Any]:
        """차트 패턴 시장 요약"""
        pattern_counts = {}
        high_confidence_patterns = []

        for stock in analyzed_stocks:
            for pattern in stock.get("chart_patterns", []):
                pattern_name = pattern["pattern_name"]
                pattern_counts[pattern_name] = pattern_counts.get(pattern_name, 0) + 1

                if pattern["confidence_score"] >= 80:
                    high_confidence_patterns.append(
                        {
                            "symbol": stock["symbol"],
                            "pattern": pattern_name,
                            "confidence": pattern["confidence_score"],
                            "expected_return": pattern["expected_return"],
                        }
                    )

        return {
            "total_patterns_detected": sum(pattern_counts.values()),
            "pattern_distribution": pattern_counts,
            "high_confidence_patterns": sorted(
                high_confidence_patterns, key=lambda x: x["confidence"], reverse=True
            )[:10],
        }

    def _summarize_triple_screen_results(
        self, analyzed_stocks: List[Dict]
    ) -> Dict[str, Any]:
        """삼중스크린 시장 요약"""
        signal_counts = {}
        strong_signals = []

        for stock in analyzed_stocks:
            triple_screen = stock.get("triple_screen", {})
            final_signal = triple_screen.get("final_signal", "관망")
            signal_counts[final_signal] = signal_counts.get(final_signal, 0) + 1

            if triple_screen.get("overall_score", 0) >= 80:
                strong_signals.append(
                    {
                        "symbol": stock["symbol"],
                        "signal": final_signal,
                        "score": triple_screen.get("overall_score", 0),
                        "entry_timing": triple_screen.get("entry_timing", ""),
                    }
                )

        return {
            "signal_distribution": signal_counts,
            "strong_signals": sorted(
                strong_signals, key=lambda x: x["score"], reverse=True
            )[:10],
        }

    def _identify_top_opportunities(
        self, analyzed_stocks: List[Dict]
    ) -> List[Dict[str, Any]]:
        """최고 투자 기회 식별"""
        opportunities = []

        for stock in analyzed_stocks:
            investment_signal = stock.get("investment_signal", {})
            signal_type = investment_signal.get("signal_type", "HOLD")
            confidence = investment_signal.get("confidence", 0)

            if signal_type in ["STRONG_BUY", "BUY"] and confidence >= 70:
                opportunity_score = confidence

                # 차트 패턴 보너스
                patterns = stock.get("chart_patterns", [])
                if patterns:
                    best_pattern = max(patterns, key=lambda x: x["confidence_score"])
                    opportunity_score += best_pattern["confidence_score"] * 0.2

                # 삼중스크린 보너스
                triple_screen = stock.get("triple_screen", {})
                if triple_screen.get("final_signal") == "강력한 매수":
                    opportunity_score += 10

                opportunities.append(
                    {
                        "symbol": stock["symbol"],
                        "company_name": stock["company_name"],
                        "current_price": stock["current_price"],
                        "signal_type": signal_type,
                        "confidence": confidence,
                        "opportunity_score": min(100, opportunity_score),
                        "key_patterns": [p["pattern_name"] for p in patterns],
                        "triple_screen_signal": triple_screen.get("final_signal", ""),
                        "analysis_summary": stock.get("gemini_summary", "")[:200]
                        + "...",
                    }
                )

        return sorted(
            opportunities, key=lambda x: x["opportunity_score"], reverse=True
        )[:20]


# Gemini AI를 위한 전용 분석 결과 생성 함수
def create_gemini_analysis_summary(analysis_results: Dict[str, Any]) -> str:
    """
    Gemini AI가 투자 결정에 활용할 수 있는 종합 분석 보고서 생성

    Args:
        analysis_results: TechnicalAnalyzer의 전체 분석 결과

    Returns:
        Gemini AI 친화적인 분석 보고서 텍스트
    """
    report = "【차트 전문가 기술적 분석 종합 보고서】\n"
    report += "=" * 50 + "\n\n"

    for market_name, market_data in analysis_results.items():
        report += f"📈 {market_name} 시장 분석:\n"
        report += f"  • 분석 종목 수: {len(market_data.get('analyzed_stocks', []))}개\n"

        # 차트 패턴 요약
        pattern_summary = market_data.get("chart_patterns_summary", {})
        report += f"  • 감지된 차트 패턴: {pattern_summary.get('total_patterns_detected', 0)}개\n"

        # 삼중스크린 요약
        triple_summary = market_data.get("triple_screen_summary", {})
        signal_dist = triple_summary.get("signal_distribution", {})
        report += f"  • 매수 신호: {signal_dist.get('강력한 매수', 0) + signal_dist.get('매수 대기', 0)}개\n"

        # 최고 기회
        top_opportunities = market_data.get("top_opportunities", [])[:5]
        if top_opportunities:
            report += f"  • Top 5 투자 기회:\n"
            for i, opp in enumerate(top_opportunities, 1):
                report += f"    {i}. {opp['symbol']} ({opp['signal_type']}, "
                report += f"신뢰도 {opp['confidence']:.0f}%)\n"

        report += "\n"

    report += "※ 이 보고서는 토마스 불코우스키의 차트 패턴 분석과 "
    report += "알렉산더 엘더의 삼중스크린 시스템을 기반으로 작성되었습니다.\n"

    return report
