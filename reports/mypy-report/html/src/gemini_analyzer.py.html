    def prepare_premium_dataset(
        self, stock_data_list: List[StockData]
    ) -> Dict[str, List[StockData]]:
        """

        고품질 데이터만 선별하여 전략별로 분류
        """
        premium_data = {"high_quality": [], "strategies": {}, "markets": {}}

        # 전략별 초기화
        for strategy in self.investment_strategies:
            premium_data["strategies"][strategy] = []

        # 시장별 초기화
        premium_data["markets"] = {
            "korean": [],  # KOSPI200
            "us_nasdaq": [],  # NASDAQ100
            "us_sp500": [],  # SP500
        }

        for stock in stock_data_list:
            # 데이터 품질 평가
            quality = self._evaluate_data_quality(stock)

            if quality.overall_score >= 75:  # 고품질 데이터만
                premium_data["high_quality"].append(stock)

                # 시장별 분류
                if stock.market == "KOSPI200":
                    premium_data["markets"]["korean"].append(stock)
                elif stock.market == "NASDAQ100":
                    premium_data["markets"]["us_nasdaq"].append(stock)
                elif stock.market == "SP500":
                    premium_data["markets"]["us_sp500"].append(stock)

                # 전략별 분류 (각 전략당 Top 15)
                for strategy in self.investment_strategies:
                    strategy_score = self._calculate_strategy_score(stock, strategy)
                    if strategy_score > 60:  # 임계값 이상만

        for strategy in self.investment_strategies:
            premium_data["strategies"][strategy] = sorted(
                premium_data["strategies"][strategy],
                key=lambda x: self._calculate_strategy_score(x, strategy),
                reverse=True,
            )[:15]

        logging.info(f"📊 프리미엄 데이터셋 준비 완료:")
        logging.info(f"   - 고품질 데이터: {len(premium_data['high_quality'])}개")

        logging.info(f"   - 미국 나스닥: {len(premium_data['markets']['us_nasdaq'])}개")
        logging.info(f"   - 미국 S&P500: {len(premium_data['markets']['us_sp500'])}개")
        for strategy, stocks in premium_data["strategies"].items():
            logging.info(f"   - {strategy}: {len(stocks)}개")

        return premium_data

    def _evaluate_data_quality(self, stock: StockData) -> DataQuality:
        """데이터 품질 평가"""
        completeness = 0
        total_fields = 0

        # 필수 필드 체크
        essential_fields = [stock.price, stock.volume, stock.market_cap]
        completeness += sum(
            1 for field in essential_fields if field is not None and field > 0
        )
        total_fields += len(essential_fields)

        # 기술적 지표 체크
        technical_fields = [
            stock.rsi,
            stock.macd,
            stock.ma_20,
            stock.ma_50,
            stock.momentum_score,
            stock.volatility,
        ]
        completeness += sum(1 for field in technical_fields if field is not None)
        total_fields += len(technical_fields)

        completeness_score = (completeness / total_fields) * 100

        # 정확성 점수 (기본값)
        accuracy_score = 85.0

        # 신선도 점수 (기본값)
        freshness_score = 90.0

        # 전체 점수
        overall_score = (
            completeness_score * 0.5 + accuracy_score * 0.3 + freshness_score * 0.2
        )

        return DataQuality(
            completeness_score=completeness_score,
            accuracy_score=accuracy_score,
            freshness_score=freshness_score,
            overall_score=overall_score,
        )

            tech_strength = 0
            if stock.rsi:
                tech_strength += min(stock.rsi / 100, 1) * 30
            if stock.macd:
                tech_strength += min(abs(stock.macd) / 5, 1) * 30
            if stock.trend_strength:
                tech_strength += stock.trend_strength * 40

            metrics["technical_strength"][stock.symbol] = tech_strength

            # 모멘텀 점수
            momentum = 0
            if stock.momentum_score:
                momentum += stock.momentum_score * 50
            if stock.ma_20 and stock.ma_50 and stock.ma_20 > stock.ma_50:
                momentum += 30
            if stock.volume and stock.volume > 1000000:
                momentum += 20

            metrics["momentum_scores"][stock.symbol] = momentum

            # 리스크 점수 (낮을수록 좋음)

            if stock.volatility:
                risk += stock.volatility * 100
            if stock.rsi and (stock.rsi > 80 or stock.rsi < 20):
                risk += 30

            metrics["risk_scores"][stock.symbol] = risk

            # 전략 일치도
            strategy_scores = {}
            for strategy in self.investment_strategies:
                strategy_scores[strategy] = self._calculate_strategy_score(
                    stock, strategy
                )
            metrics["strategy_alignment"][stock.symbol] = strategy_scores

        return metrics

    def calculate_precise_entry_target(
        self, stock: StockData, strategy: str
    ) -> Dict[str, float]:
        """
        차트 전문가 전략별 정밀한 진입가격과 목표가격 계산
        토마스 불코우스키와 알렉산더 엘더의 타이밍 기법 적용
        """
        result = {
            "entry_price": stock.price,
            "target_price": stock.price * 1.1,
            "stop_loss": stock.price * 0.95,
            "expected_return_pct": 10.0,
            "confidence_level": 0.7,
            "timing_score": 50.0,
        }

        if strategy == "thomas_bulkowski":
            # 불코우스키의 차트 패턴 기반 타이밍

            # 1. 저항선 돌파 기반 진입가격
            if stock.resistance_level and stock.price:
                resistance_ratio = stock.price / stock.resistance_level

                if 0.98 <= resistance_ratio <= 1.02:  # 저항선 근접
                    # 저항선 돌파 대기 전략
                    result["entry_price"] = (
                        stock.resistance_level * 1.015
                    )  # 1.5% 돌파시 진입
                    result["target_price"] = stock.resistance_level * 1.15  # 15% 목표
                    result["stop_loss"] = stock.resistance_level * 0.97  # 3% 손절
                    result["expected_return_pct"] = 13.0
                    result["confidence_level"] = 0.85
                    result["timing_score"] = 85.0

                elif 1.02 < resistance_ratio <= 1.05:  # 돌파 초기
                    # 돌파 확인 후 추격 진입
                    result["entry_price"] = stock.price * 0.995  # 현재가 근처 진입
                    result["target_price"] = stock.resistance_level * 1.12  # 12% 목표
                    result["stop_loss"] = (
                        stock.resistance_level * 0.985
                    )  # 저항선 아래 손절
                    result["expected_return_pct"] = 10.5
                    result["confidence_level"] = 0.80
                    result["timing_score"] = 80.0

                elif resistance_ratio > 1.05:  # 강한 돌파
                    # 재조정 대기 후 진입
                    result["entry_price"] = stock.price * 0.97  # 3% 조정시 진입
                    result["target_price"] = stock.price * 1.18  # 18% 목표
                    result["stop_loss"] = (
                        stock.resistance_level * 0.98
                    )  # 저항선 근처 손절
                    result["expected_return_pct"] = 16.0
                    result["confidence_level"] = 0.75
                    result["timing_score"] = 75.0

            # 2. 볼린저 밴드 기반 타이밍 조정
            if stock.bollinger_upper and stock.bollinger_lower and stock.price:
                bb_position = (stock.price - stock.bollinger_lower) / (
                    stock.bollinger_upper - stock.bollinger_lower
                )

                if bb_position > 0.8:  # 상단 근처 - 조심스러운 진입
                    result["entry_price"] = stock.price * 0.98
                    result["confidence_level"] *= 0.9
                elif bb_position < 0.2:  # 하단 근처 - 적극적 진입
                    result["entry_price"] = stock.price * 1.01
                    result["confidence_level"] *= 1.1

            # 3. 거래량 확인으로 신뢰도 조정
            if stock.volume:
                if stock.volume > 2000000:  # 고거래량
                    result["confidence_level"] *= 1.15
                    result["timing_score"] += 10
                elif stock.volume < 500000:  # 저거래량
                    result["confidence_level"] *= 0.85
                    result["timing_score"] -= 10

        elif strategy == "alexander_elder":
            # 엘더의 삼중스크린 기반 정밀 타이밍

            timing_signals = []

            # 1차 스크린: 장기 추세 확인
            long_term_signal = 0
            if stock.ma_200 and stock.price:
                if stock.price > stock.ma_200:
                    distance_ratio = (stock.price - stock.ma_200) / stock.ma_200
                    if 0.02 <= distance_ratio <= 0.10:  # 2-10% 위
                        long_term_signal = 1
                        timing_signals.append("장기추세양호")
                    elif distance_ratio > 0.15:  # 15% 이상 - 과열
                        long_term_signal = -1
                        timing_signals.append("장기과열주의")

            # 2차 스크린: 중기 오실레이터
            medium_term_signal = 0
            if stock.rsi:
                if 35 <= stock.rsi <= 50:  # 이상적 진입 구간
                    medium_term_signal = 1
                    timing_signals.append("RSI최적구간")
                elif 50 < stock.rsi <= 65:  # 양호한 구간
                    medium_term_signal = 0.5
                    timing_signals.append("RSI양호구간")
                elif stock.rsi > 70:  # 과매수
                    medium_term_signal = -1
                    timing_signals.append("RSI과매수")

            # 3차 스크린: 단기 진입 타이밍
            short_term_signal = 0
            if stock.ma_20 and stock.price:
                ma20_distance = (stock.price - stock.ma_20) / stock.ma_20
                if -0.02 <= ma20_distance <= 0.03:  # MA20 근처
                    short_term_signal = 1
                    timing_signals.append("단기진입타이밍")
                elif ma20_distance > 0.05:  # 5% 이상 위
                    short_term_signal = -0.5
                    timing_signals.append("단기과열")

            # 종합 신호 계산
            total_signal = long_term_signal + medium_term_signal + short_term_signal

            if total_signal >= 2:  # 강한 매수 신호
                result["entry_price"] = stock.price * 1.005  # 현재가 근처 진입
                result["target_price"] = stock.price * 1.12  # 12% 목표
                result["stop_loss"] = (
                    stock.ma_20 * 0.98 if stock.ma_20 else stock.price * 0.95
                )
                result["expected_return_pct"] = 11.5
                result["confidence_level"] = 0.88
                result["timing_score"] = 88.0

            elif total_signal >= 1:  # 중간 매수 신호
                result["entry_price"] = stock.price * 0.98  # 2% 조정 대기
                result["target_price"] = stock.price * 1.08  # 8% 목표
                result["stop_loss"] = (
                    stock.ma_20 * 0.97 if stock.ma_20 else stock.price * 0.94
                )
                result["expected_return_pct"] = 8.5
                result["confidence_level"] = 0.75
                result["timing_score"] = 75.0

            elif total_signal <= -1:  # 약한 신호 또는 위험
                result["entry_price"] = stock.price * 0.95  # 5% 조정 대기
                result["target_price"] = stock.price * 1.05  # 보수적 목표
                result["stop_loss"] = stock.price * 0.92
                result["expected_return_pct"] = 5.0
                result["confidence_level"] = 0.60
                result["timing_score"] = 60.0

            # MACD 확인으로 타이밍 조정
            if stock.macd:
                if stock.macd > 0 and stock.macd < 2:  # 양수이지만 과열 아님
                    result["confidence_level"] *= 1.1
                    result["timing_score"] += 5
                elif stock.macd < 0:  # 음수 MACD
                    result["confidence_level"] *= 0.9
                    result["timing_score"] -= 5

        # 최종 범위 조정
        result["confidence_level"] = min(max(result["confidence_level"], 0.3), 1.0)
        result["timing_score"] = min(max(result["timing_score"], 30.0), 100.0)
        result["expected_return_pct"] = round(result["expected_return_pct"], 1)

        return result

            sample_stocks = premium_data["high_quality"][:10]  # 상위 10개 종목

            bulkowski_signals = []

            for stock in sample_stocks:

                    bulkowski_timing = self.calculate_precise_entry_target(
                        stock, "thomas_bulkowski"

                            {
                                "symbol": stock.symbol,
                                "timing_score": bulkowski_timing["timing_score"],

                    # 엘더 타이밍 분석

                        stock, "alexander_elder"
                    )
                    if elder_timing["timing_score"] > 75:
                        elder_signals.append(
                            {
                                "symbol": stock.symbol,
                                "timing_score": elder_timing["timing_score"],
                                "confidence": elder_timing["confidence_level"],
                            }
                        )

                except Exception as e:
                    continue

            # 불코우스키 분석 요약

                avg_bulkowski_score = sum(
                    s["timing_score"] for s in bulkowski_signals

            if elder_signals:
                avg_elder_score = sum(s["timing_score"] for s in elder_signals) / len(
                    elder_signals
                )
                timing_summary.append(
                    f"🔹 **엘더 삼중스크린 분석**: {len(elder_signals)}개 종목에서 삼중스크린 진입 신호 확인 (평균 타이밍 점수: {avg_elder_score:.1f})"
                )

        # 전체 시장 타이밍 평가
        timing_summary.append(
            "🔹 **현재 시장 타이밍**: 차트 전문가 관점에서 선별적 진입 기회 존재"
        )

        )

        return (
            "\n".join(timing_summary)
            if timing_summary
            else "차트 전문가 타이밍 분석 진행 중..."
        )
