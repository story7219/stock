@dataclass
class CacheEntry:
    """Ï∫êÏãú ÏóîÌä∏Î¶¨ ÌÅ¥ÎûòÏä§"""

    data: Any
    timestamp: datetime
    ttl: int  # seconds
    data_hash: str
    access_count: int = 0
    last_access: datetime = None

    def is_expired(self) -> bool:
        """Ï∫êÏãú ÎßåÎ£å Ïó¨Î∂Ä ÌôïÏù∏"""
        return (datetime.now() - self.timestamp).total_seconds() > self.ttl

    def is_valid(self) -> bool:
        """Ï∫êÏãú Ïú†Ìö®ÏÑ± ÌôïÏù∏"""
        return not self.is_expired()

class SmartDataStorage:
    """Ïä§ÎßàÌä∏ Îç∞Ïù¥ÌÑ∞ Ï†ÄÏû•ÏÜå - Íµ¨Í∏ÄÏãúÌä∏ Í∏∞Î∞ò"""

    def __init__(self):
        """Ï¥àÍ∏∞Ìôî"""
        self.logger = self._setup_logger()

        # Ï∫êÏãú ÏÑ§Ï†ï
        self.cache_dir = Path("data_cache")
        self.cache_dir.mkdir(exist_ok=True)

        # Î©îÎ™®Î¶¨ Ï∫êÏãú
        self.memory_cache: Dict[str, CacheEntry] = {}
        self.max_memory_cache_size = 100  # ÏµúÎåÄ 100Í∞ú Ìï≠Î™©

        # ÎîîÏä§ÌÅ¨ Ï∫êÏãú
        self.disk_cache_dir = self.cache_dir / "disk_cache"
        self.disk_cache_dir.mkdir(exist_ok=True)

        # Íµ¨Í∏Ä ÏãúÌä∏ ÏÑ§Ï†ï
        self.credentials_path = os.getenv(
            "GOOGLE_SHEETS_CREDENTIALS_PATH", "credentials.json"
        )
        self.spreadsheet_id = os.getenv("GOOGLE_SHEETS_SPREADSHEET_ID")

        # Ï∫êÏãú TTL ÏÑ§Ï†ï (Ï¥à)
        self.cache_ttl = {
            "stock_data": 300,  # 5Î∂Ñ
            "analysis_results": 1800,  # 30Î∂Ñ
            "market_data": 600,  # 10Î∂Ñ
            "news_data": 900,  # 15Î∂Ñ
            "historical_data": 86400,  # 24ÏãúÍ∞Ñ
            "dashboard_data": 60,  # 1Î∂Ñ
        }

        # Íµ¨Í∏Ä ÏãúÌä∏ ÌÅ¥ÎùºÏù¥Ïñ∏Ìä∏
        self.sheets_client = None
        self.spreadsheet = None
        self.executor = ThreadPoolExecutor(max_workers=5)

        # ÌÜµÍ≥Ñ
        self.stats = {
            "cache_hits": 0,
            "cache_misses": 0,
            "sheets_reads": 0,
            "sheets_writes": 0,
            "data_efficiency": 0.0,
        }

        self._initialize_sheets_client()
        self._load_disk_cache_index()

        self.logger.info("üöÄ Smart Data Storage Ï¥àÍ∏∞Ìôî ÏôÑÎ£å")

    def _setup_logger(self) -> logging.Logger:
        """Î°úÍ±∞ ÏÑ§Ï†ï"""
        logger = logging.getLogger("SmartDataStorage")
        logger.setLevel(logging.INFO)

        if not logger.handlers:
            # Î°úÍ∑∏ ÎîîÎ†âÌÜ†Î¶¨ ÏÉùÏÑ±
            os.makedirs("logs", exist_ok=True)

            console_handler = logging.StreamHandler()

            logger.addHandler(console_handler)

        return logger

    def _initialize_sheets_client(self):
        """Íµ¨Í∏Ä ÏãúÌä∏ ÌÅ¥ÎùºÏù¥Ïñ∏Ìä∏ Ï¥àÍ∏∞Ìôî"""
        try:
            if not self.credentials_path or not os.path.exists(self.credentials_path):
                self.logger.warning("‚ö†Ô∏è Íµ¨Í∏Ä Ïù∏Ï¶ù ÌååÏùºÏù¥ ÏóÜÏäµÎãàÎã§")
                return

            if not self.spreadsheet_id:
                self.logger.warning("‚ö†Ô∏è Ïä§ÌîÑÎ†àÎìúÏãúÌä∏ IDÍ∞Ä ÏÑ§Ï†ïÎêòÏßÄ ÏïäÏïòÏäµÎãàÎã§")
                return

            # Ïù∏Ï¶ù ÏÑ§Ï†ï
            scopes = [
                "https://www.googleapis.com/auth/spreadsheets",
                "https://www.googleapis.com/auth/drive",
            ]

            credentials = Credentials.from_service_account_file(
                self.credentials_path, scopes=scopes
            )

            self.sheets_client = gspread.authorize(credentials)
            self.spreadsheet = self.sheets_client.open_by_key(self.spreadsheet_id)

            self.logger.info("‚úÖ Íµ¨Í∏Ä ÏãúÌä∏ ÌÅ¥ÎùºÏù¥Ïñ∏Ìä∏ Ï¥àÍ∏∞Ìôî ÏôÑÎ£å")

        except Exception as e:
            self.logger.error(f"‚ùå Íµ¨Í∏Ä ÏãúÌä∏ Ï¥àÍ∏∞Ìôî Ïã§Ìå®: {e}")
            self.sheets_client = None
            self.spreadsheet = None

    def _load_disk_cache_index(self):
        """ÎîîÏä§ÌÅ¨ Ï∫êÏãú Ïù∏Îç±Ïä§ Î°úÎìú"""
        try:
            index_file = self.disk_cache_dir / "cache_index.json"
            if index_file.exists():
                with open(index_file, "r", encoding="utf-8") as f:
                    self.disk_cache_index = json.load(f)
            else:
                self.disk_cache_index = {}
        except Exception as e:
            self.logger.warning(f"‚ö†Ô∏è ÎîîÏä§ÌÅ¨ Ï∫êÏãú Ïù∏Îç±Ïä§ Î°úÎìú Ïã§Ìå®: {e}")
            self.disk_cache_index = {}

    def _save_disk_cache_index(self):
        """ÎîîÏä§ÌÅ¨ Ï∫êÏãú Ïù∏Îç±Ïä§ Ï†ÄÏû•"""
        try:
            index_file = self.disk_cache_dir / "cache_index.json"
            with open(index_file, "w", encoding="utf-8") as f:
                json.dump(self.disk_cache_index, f, ensure_ascii=False, indent=2)
        except Exception as e:
            self.logger.warning(f"‚ö†Ô∏è ÎîîÏä§ÌÅ¨ Ï∫êÏãú Ïù∏Îç±Ïä§ Ï†ÄÏû• Ïã§Ìå®: {e}")

    def _generate_cache_key(self, data_type: str, parameters: Dict[str, Any]) -> str:
        """Ï∫êÏãú ÌÇ§ ÏÉùÏÑ±"""
        key_data = f"{data_type}:{json.dumps(parameters, sort_keys=True)}"
        return hashlib.md5(key_data.encode()).hexdigest()

    def _calculate_data_hash(self, data: Any) -> str:
        """Îç∞Ïù¥ÌÑ∞ Ìï¥Ïãú Í≥ÑÏÇ∞"""
        data_str = json.dumps(data, sort_keys=True, default=str)
        return hashlib.sha256(data_str.encode()).hexdigest()

    async def get_data(
        self,
        data_type: str,
        parameters: Dict[str, Any] = None,
        force_refresh: bool = False,
    ) -> Optional[Any]:
        """Ïä§ÎßàÌä∏ Îç∞Ïù¥ÌÑ∞ Ï°∞Ìöå - Ï∫êÏãú Ïö∞ÏÑ†, ÌïÑÏöîÏãú Íµ¨Í∏ÄÏãúÌä∏ÏóêÏÑú Ï°∞Ìöå"""
        if parameters is None:
            parameters = {}

        cache_key = self._generate_cache_key(data_type, parameters)

        # Í∞ïÏ†ú ÏÉàÎ°úÍ≥†Ïπ®Ïù¥ ÏïÑÎãå Í≤ΩÏö∞ Ï∫êÏãú ÌôïÏù∏
        if not force_refresh:
            # 1. Î©îÎ™®Î¶¨ Ï∫êÏãú ÌôïÏù∏
            if cache_key in self.memory_cache:
                entry = self.memory_cache[cache_key]
                if entry.is_valid():
                    entry.access_count += 1
                    entry.last_access = datetime.now()
                    self.stats["cache_hits"] += 1
                    self.logger.debug(f"üíæ Î©îÎ™®Î¶¨ Ï∫êÏãú ÌûàÌä∏: {data_type}")
                    return entry.data
                else:
                    # ÎßåÎ£åÎêú Ï∫êÏãú Ï†úÍ±∞
                    del self.memory_cache[cache_key]

            # 2. ÎîîÏä§ÌÅ¨ Ï∫êÏãú ÌôïÏù∏
            disk_data = await self._load_from_disk_cache(cache_key, data_type)
            if disk_data is not None:
                # Î©îÎ™®Î¶¨ Ï∫êÏãúÏóêÎèÑ Ï†ÄÏû•
                self._store_in_memory_cache(cache_key, disk_data, data_type)
                self.stats["cache_hits"] += 1
                self.logger.debug(f"üíø ÎîîÏä§ÌÅ¨ Ï∫êÏãú ÌûàÌä∏: {data_type}")
                return disk_data

        # 3. Íµ¨Í∏Ä ÏãúÌä∏ÏóêÏÑú Îç∞Ïù¥ÌÑ∞ Ï°∞Ìöå
        self.stats["cache_misses"] += 1
        self.logger.info(f"üîç Íµ¨Í∏ÄÏãúÌä∏ÏóêÏÑú Îç∞Ïù¥ÌÑ∞ Ï°∞Ìöå: {data_type}")

        sheet_data = await self._fetch_from_sheets(data_type, parameters)

        if sheet_data is not None:
            # Ï∫êÏãúÏóê Ï†ÄÏû•
            await self._store_data(cache_key, sheet_data, data_type)
            self.stats["sheets_reads"] += 1
            return sheet_data

        return None

    async def store_data(
        self, data_type: str, data: Any, parameters: Dict[str, Any] = None
    ) -> bool:
        """Îç∞Ïù¥ÌÑ∞ Ï†ÄÏû• - Íµ¨Í∏ÄÏãúÌä∏ + Ï∫êÏãú"""
        if parameters is None:
            parameters = {}

        try:
            # 1. Íµ¨Í∏Ä ÏãúÌä∏Ïóê Ï†ÄÏû•
            success = await self._save_to_sheets(data_type, data, parameters)

            if success:
                # 2. Ï∫êÏãúÏóêÎèÑ Ï†ÄÏû•
                cache_key = self._generate_cache_key(data_type, parameters)
                await self._store_data(cache_key, data, data_type)
                self.stats["sheets_writes"] += 1

                self.logger.info(f"‚úÖ Îç∞Ïù¥ÌÑ∞ Ï†ÄÏû• ÏôÑÎ£å: {data_type}")
                return True

            return False

        except Exception as e:
            self.logger.error(f"‚ùå Îç∞Ïù¥ÌÑ∞ Ï†ÄÏû• Ïã§Ìå®: {e}")
            return False

    async def _fetch_from_sheets(
        self, data_type: str, parameters: Dict[str, Any]
    ) -> Optional[Any]:
        """Íµ¨Í∏Ä ÏãúÌä∏ÏóêÏÑú Îç∞Ïù¥ÌÑ∞ Ï°∞Ìöå"""
        if not self.spreadsheet:
            return None

        try:
            # Îç∞Ïù¥ÌÑ∞ ÌÉÄÏûÖÎ≥Ñ ÏãúÌä∏ Îß§Ìïë
            sheet_mapping = {
                "stock_data": "Ï£ºÏãùÎç∞Ïù¥ÌÑ∞",
                "analysis_results": "AIÎ∂ÑÏÑùÍ≤∞Í≥º",
                "korean_market_top5": "ÌïúÍµ≠ÏãúÏû•TOP5",
                "us_market_top5": "ÎØ∏Íµ≠ÏãúÏû•TOP5",
                "strategy_summary": "Ï†ÑÎûµÏöîÏïΩ",
                "master_recommendation": "ÎßàÏä§ÌÑ∞Ï∂îÏ≤ú",
                "daily_summary": "ÏùºÏùºÏöîÏïΩ",
                "dashboard_data": "ÎåÄÏãúÎ≥¥Îìú",
            }

            sheet_name = sheet_mapping.get(data_type)
            if not sheet_name:
                return None

            # ÎπÑÎèôÍ∏∞Î°ú ÏãúÌä∏ Îç∞Ïù¥ÌÑ∞ Ï°∞Ìöå
            loop = asyncio.get_event_loop()
            worksheet = await loop.run_in_executor(
                self.executor, self.spreadsheet.worksheet, sheet_name
            )

            # ÌååÎùºÎØ∏ÌÑ∞Ïóê Îî∞Î•∏ ÌïÑÌÑ∞ÎßÅ
            if parameters.get("limit"):
                records = await loop.run_in_executor(
                    self.executor,
                    lambda: worksheet.get_all_records()[: parameters["limit"]],
                )
            elif parameters.get("date_from"):
                # ÎÇ†Ïßú Î≤îÏúÑ ÌïÑÌÑ∞ÎßÅ
                all_records = await loop.run_in_executor(
                    self.executor, worksheet.get_all_records
                )

                date_from = datetime.strptime(parameters["date_from"], "%Y-%m-%d")
                records = [
                    record
                    for record in all_records
                    if datetime.strptime(record.get("ÎÇ†Ïßú", "1900-01-01"), "%Y-%m-%d")
                    >= date_from
                ]
            else:
                records = await loop.run_in_executor(
                    self.executor, worksheet.get_all_records
                )

            return records

        except Exception as e:
            self.logger.error(f"‚ùå Íµ¨Í∏ÄÏãúÌä∏ Ï°∞Ìöå Ïã§Ìå®: {e}")
            return None

    async def _save_to_sheets(
        self, data_type: str, data: Any, parameters: Dict[str, Any]
    ) -> bool:
        """Íµ¨Í∏Ä ÏãúÌä∏Ïóê Îç∞Ïù¥ÌÑ∞ Ï†ÄÏû•"""

            elif data_type == "analysis_results":

            return True

        except Exception as e:
            self.logger.error(f"‚ùå Íµ¨Í∏ÄÏãúÌä∏ Ï†ÄÏû• Ïã§Ìå®: {e}")

    def _store_in_memory_cache(self, cache_key: str, data: Any, data_type: str):
        """Î©îÎ™®Î¶¨ Ï∫êÏãúÏóê Ï†ÄÏû•"""
        try:
            # Î©îÎ™®Î¶¨ Ï∫êÏãú ÌÅ¨Í∏∞ Ï†úÌïú
            if len(self.memory_cache) >= self.max_memory_cache_size:
                # LRU Î∞©ÏãùÏúºÎ°ú Ïò§ÎûòÎêú Ìï≠Î™© Ï†úÍ±∞
                oldest_key = min(
                    self.memory_cache.keys(),
                    key=lambda k: self.memory_cache[k].last_access or datetime.min,
                )
                del self.memory_cache[oldest_key]

            ttl = self.cache_ttl.get(data_type, 300)
            data_hash = self._calculate_data_hash(data)

            entry = CacheEntry(
                data=data,
                timestamp=datetime.now(),
                ttl=ttl,
                data_hash=data_hash,
                access_count=1,
                last_access=datetime.now(),
            )

            self.memory_cache[cache_key] = entry

        except Exception as e:
            self.logger.warning(f"‚ö†Ô∏è Î©îÎ™®Î¶¨ Ï∫êÏãú Ï†ÄÏû• Ïã§Ìå®: {e}")

    async def _load_from_disk_cache(
        self, cache_key: str, data_type: str
    ) -> Optional[Any]:
        """ÎîîÏä§ÌÅ¨ Ï∫êÏãúÏóêÏÑú Î°úÎìú"""
        try:
            cache_file = self.disk_cache_dir / f"{cache_key}.pkl"

            if not cache_file.exists():
                return None

            # ÌååÏùº ÏàòÏ†ï ÏãúÍ∞Ñ ÌôïÏù∏
            file_mtime = datetime.fromtimestamp(cache_file.stat().st_mtime)
            ttl = self.cache_ttl.get(data_type, 300)

            if (datetime.now() - file_mtime).total_seconds() > ttl:
                # ÎßåÎ£åÎêú Ï∫êÏãú ÌååÏùº ÏÇ≠Ï†ú
                cache_file.unlink()
                return None

            # Îç∞Ïù¥ÌÑ∞ Î°úÎìú
            loop = asyncio.get_event_loop()
            data = await loop.run_in_executor(
                self.executor, self._load_pickle_file, cache_file
            )

            return data

        except Exception as e:
            self.logger.warning(f"‚ö†Ô∏è ÎîîÏä§ÌÅ¨ Ï∫êÏãú Î°úÎìú Ïã§Ìå®: {e}")
            return None

    def _load_pickle_file(self, file_path: Path) -> Any:
        """ÌîºÌÅ¥ ÌååÏùº Î°úÎìú (ÎèôÍ∏∞ Ìï®Ïàò)"""
        with open(file_path, "rb") as f:
            return pickle.load(f)

    async def _store_data(self, cache_key: str, data: Any, data_type: str):
        """Ï∫êÏãúÏóê Îç∞Ïù¥ÌÑ∞ Ï†ÄÏû• (Î©îÎ™®Î¶¨ + ÎîîÏä§ÌÅ¨)"""
        # Î©îÎ™®Î¶¨ Ï∫êÏãú
        self._store_in_memory_cache(cache_key, data, data_type)

        # ÎîîÏä§ÌÅ¨ Ï∫êÏãú
        await self._store_in_disk_cache(cache_key, data, data_type)

    async def _store_in_disk_cache(self, cache_key: str, data: Any, data_type: str):
        """ÎîîÏä§ÌÅ¨ Ï∫êÏãúÏóê Ï†ÄÏû•"""
        try:
            cache_file = self.disk_cache_dir / f"{cache_key}.pkl"

            loop = asyncio.get_event_loop()
            await loop.run_in_executor(
                self.executor, self._save_pickle_file, cache_file, data
            )

            # Ïù∏Îç±Ïä§ ÏóÖÎç∞Ïù¥Ìä∏
            self.disk_cache_index[cache_key] = {
                "data_type": data_type,
                "timestamp": datetime.now().isoformat(),
                "file_path": str(cache_file),
            }

            self._save_disk_cache_index()

        except Exception as e:
            self.logger.warning(f"‚ö†Ô∏è ÎîîÏä§ÌÅ¨ Ï∫êÏãú Ï†ÄÏû• Ïã§Ìå®: {e}")

    def _save_pickle_file(self, file_path: Path, data: Any):
        """ÌîºÌÅ¥ ÌååÏùº Ï†ÄÏû• (ÎèôÍ∏∞ Ìï®Ïàò)"""
        with open(file_path, "wb") as f:
            pickle.dump(data, f)

    def get_cache_stats(self) -> Dict[str, Any]:
        """Ï∫êÏãú ÌÜµÍ≥Ñ Ï°∞Ìöå"""
        total_requests = self.stats["cache_hits"] + self.stats["cache_misses"]
        hit_rate = (
            (self.stats["cache_hits"] / total_requests * 100)
            if total_requests > 0
            else 0
        )

        return {
            "cache_hits": self.stats["cache_hits"],
            "cache_misses": self.stats["cache_misses"],
            "hit_rate": f"{hit_rate:.1f}%",
            "sheets_reads": self.stats["sheets_reads"],
            "sheets_writes": self.stats["sheets_writes"],
            "memory_cache_size": len(self.memory_cache),
            "disk_cache_size": len(self.disk_cache_index),
            "efficiency_score": hit_rate,
        }

    async def cleanup_cache(self, max_age_hours: int = 24):
        """Ï∫êÏãú Ï†ïÎ¶¨"""
        try:
            cutoff_time = datetime.now() - timedelta(hours=max_age_hours)

            # Î©îÎ™®Î¶¨ Ï∫êÏãú Ï†ïÎ¶¨
            expired_keys = [
                key
                for key, entry in self.memory_cache.items()
                if entry.timestamp < cutoff_time
            ]

            for key in expired_keys:
                del self.memory_cache[key]

            # ÎîîÏä§ÌÅ¨ Ï∫êÏãú Ï†ïÎ¶¨
            expired_files = []
            for cache_key, info in self.disk_cache_index.items():
                timestamp = datetime.fromisoformat(info["timestamp"])
                if timestamp < cutoff_time:
                    file_path = Path(info["file_path"])
                    if file_path.exists():
                        file_path.unlink()
                    expired_files.append(cache_key)

            for key in expired_files:
                del self.disk_cache_index[key]

            self._save_disk_cache_index()

            self.logger.info(
                f"üßπ Ï∫êÏãú Ï†ïÎ¶¨ ÏôÑÎ£å: Î©îÎ™®Î¶¨ {len(expired_keys)}Í∞ú, ÎîîÏä§ÌÅ¨ {len(expired_files)}Í∞ú ÏÇ≠Ï†ú"
            )

        except Exception as e:
            self.logger.error(f"‚ùå Ï∫êÏãú Ï†ïÎ¶¨ Ïã§Ìå®: {e}")

    async def prefetch_data(self, data_requests: List[DataRequest]):
        """Îç∞Ïù¥ÌÑ∞ ÎØ∏Î¶¨ Í∞ÄÏ†∏Ïò§Í∏∞ (Î∞±Í∑∏ÎùºÏö¥Îìú)"""
        try:
            # Ïö∞ÏÑ†ÏàúÏúÑÎ≥Ñ Ï†ïÎ†¨
            sorted_requests = sorted(data_requests, key=lambda x: x.priority)

            tasks = []
            for request in sorted_requests:
                task = self.get_data(request.query_type, request.parameters)
                tasks.append(task)

            # Î≥ëÎ†¨ Ïã§Ìñâ
            await asyncio.gather(*tasks, return_exceptions=True)

            self.logger.info(f"üöÄ Îç∞Ïù¥ÌÑ∞ ÌîÑÎ¶¨ÌéòÏπò ÏôÑÎ£å: {len(data_requests)}Í∞ú ÏöîÏ≤≠")

        except Exception as e:
            self.logger.error(f"‚ùå Îç∞Ïù¥ÌÑ∞ ÌîÑÎ¶¨ÌéòÏπò Ïã§Ìå®: {e}")

    async def close(self):
        """Î¶¨ÏÜåÏä§ Ï†ïÎ¶¨"""
        try:
            # Ïä§Î†àÎìúÌíÄ Ï¢ÖÎ£å
            self.executor.shutdown(wait=True)

            # Ï∫êÏãú Ïù∏Îç±Ïä§ Ï†ÄÏû•
            self._save_disk_cache_index()

            self.logger.info("‚úÖ Smart Data Storage Ï¢ÖÎ£å ÏôÑÎ£å")

        except Exception as e:
            self.logger.error(f"‚ùå Î¶¨ÏÜåÏä§ Ï†ïÎ¶¨ Ïã§Ìå®: {e}")
