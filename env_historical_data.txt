# 한투 API + DART API 과거 데이터 수집 환경변수 설정
# 이 파일을 .env로 복사하고 실제 API 키로 교체하세요

# 한투 Open API 설정 (모의투자)
KIS_APP_KEY=your_kis_app_key_here
KIS_APP_SECRET=your_kis_app_secret_here

# 한투 Open API 설정 (실전투자)
KIS_REAL_APP_KEY=your_kis_real_app_key_here
KIS_REAL_APP_SECRET=your_kis_real_app_secret_here

# DART API 설정
DART_API_KEY=your_dart_api_key_here

# API 키 발급 방법:
# 
# 1. 한투 Open API 키 발급:
#    - https://securities.koreainvestment.com 접속
#    - Open API > API 신청
#    - 모의투자/실전투자 각각 신청 필요
#
# 2. DART API 키 발급:
#    - https://opendart.fss.or.kr 접속
#    - 회원가입 후 API 키 발급
#
# 3. 환경변수 설정:
#    - Windows: set KIS_APP_KEY=your_key
#    - Linux/Mac: export KIS_APP_KEY=your_key
#    - 또는 .env 파일 사용 (python-dotenv 설치 필요)
#
# 4. 실행:
#    python run_historical_data_collection.py
